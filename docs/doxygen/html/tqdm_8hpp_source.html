282 void print_bar(std::stringstream &ss, double filled) const {
283 auto num_filled = static_cast<index>(std::round(filled * bar_size_));
284 static const char symbols[] = {'!', '@', '$', '%', '^', '&', '*', '+', '~'};
285 int num_symbols = sizeof(symbols) / sizeof(symbols[0]);
286
287 ss << '[';
288 for (index i = 0; i < bar_size_; ++i) {
289 if (i < num_filled) {
290 ss << '#';
291 } else if (i == num_filled && num_filled > 0) {
292 ss << symbols[symbol_index_ % num_symbols]; // Use symbol_index_ for frequency
293 } else {
294 ss << '.';
295 }
296 }
297 ss << ']';
298 }
